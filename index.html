Script de Python para Análisis de Datos:

import pandas as pd
import google.generativeai as genai

# Configura la API Key (se recomienda usar variables de entorno)
genai.configure(api_key="YOUR_API_KEY")

def enriquecer_datos_cliente(dataframe):
    model = genai.GenerativeModel('gemini-pro')
    # Genera un resumen de comportamiento para cada cliente
    dataframe['resumen_ia'] = dataframe['historial_compra'].apply(
        lambda x: model.generate_content(f"Resume el patrón de compra: {x}").text       
    )
    return dataframe
